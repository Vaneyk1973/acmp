class AT1{
    int[] a;
    public AT1(int n){
        a=new int[n];
    }
    public void fillArray(int rand, int n){
        Random random = new Random();
        if (n<a.length)
            for (int i = 0; i < n; i++) {
                a[i] = random.nextInt(rand);
            }
    }
    //TODO напишите метод линейного поиска элемента num в массиве
    public int linearSearch(int num){
        for (int i=0;i<a.length;i++){
            if (a[i]==num){
                return i;
            }
        }
        return -1;
    }
}

class AT2{
    int[] a;
    public AT2(int n){
        a=new int[n];
    }

    public void fillArray(int rand, int n){
        Random random = new Random();
        if (n<a.length)
            for (int i = 0; i < n; i++) {
                a[i] = random.nextInt(rand);
            }
    }
    //TODO напишите метод сортировки массива (можно использовать метод класса Arrays)
    public void sort(int from, int to){
        Arrays.sort(a, from, to);
    }

    //TODO напишите метод бинарного поиска элемента num в массиве
    public int binarySearch(int num){
        Arrays.sort(a);
        int flag=a.length/2;
        while (true){
            if (a[flag]<num){
                flag=flag+(a.length-flag)/2;
            }
            else if (a[flag]>num)
                flag=flag/2;
            else return flag;
        }
    }
}
