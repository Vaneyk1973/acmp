import java.io.PrintStream;
import java.util.Scanner;

public class Main {

	public static Scanner sc=new Scanner(System.in);
	public static PrintStream out=System.out;

	public static void main(String[] args) {
		int a=sc.nextInt(), b[]=new int[a];
		for (int i=0;i<a;i++)
		{
			b[i]=sc.nextInt();
		}
		for (int i=a-1;i>=0;i--)
		{
			out.print(b[i]+" ");
		}
	}
}
